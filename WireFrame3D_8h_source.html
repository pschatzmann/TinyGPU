20/**
21 * @brief Lightweight 3D wireframe renderer for TinyGPU drawing targets.
22 *
23 * WireFrame3D implements a compact CPU-side 3D pipeline intended for embedded
24 * and Arduino-style environments where a full 3D engine would be excessive.
25 * It stores no scene graph of its own; instead, it provides the math and
26 * rendering steps needed to transform user-supplied mesh data into 2D line
27 * output on any ISurface-compatible surface.
28 *
29 * The class covers the main stages of a simple wireframe renderer:
30 * - 3D vector and 4x4 matrix operations
31 * - model transforms such as translation, scaling, and rotation
32 * - camera setup via a look-at style view transform
33 * - perspective or orthographic projection into screen space
34 * - minimal per-pixel depth handling for overlapping wireframe edges
35 *
36 * Meshes are represented as vertices plus edge lists, which keeps memory usage
37 * predictable and makes the class suitable for simple geometry such as cubes,
38 * spheres, axes, grids, and custom debug or UI visualizations.
39 */

365 /// Creates a wireframe grid on the XZ plane.
366 static Mesh grid(size_t subdivisions = 10, float spacing = 1.0f) {
367 Mesh mesh;
368 const float halfExtent =
369 (static_cast<float>(subdivisions) * spacing) * 0.5f;
370
371 for (size_t index = 0; index <= subdivisions; ++index) {
372 const float offset = (static_cast<float>(index) * spacing) - halfExtent;
373
374 const size_t lineStart = mesh.vertices.size();
375 mesh.vertices.push_back({-halfExtent, 0.0f, offset});
376 mesh.vertices.push_back({halfExtent, 0.0f, offset});
377 mesh.edges.push_back({lineStart, lineStart + 1});
378
379 const size_t columnStart = mesh.vertices.size();
380 mesh.vertices.push_back({offset, 0.0f, -halfExtent});
381 mesh.vertices.push_back({offset, 0.0f, halfExtent});
382 mesh.edges.push_back({columnStart, columnStart + 1});
383 }
384
385 return mesh;
386 }

388 /// Creates a wireframe sphere centered at the origin.
389 static Mesh sphere(float radius = 1.0f, size_t latitudeSteps = 8,
390 size_t longitudeSteps = 12) {
391 Mesh mesh;
392 const size_t latCount = std::max<size_t>(2, latitudeSteps);
393 const size_t lonCount = std::max<size_t>(3, longitudeSteps);
394 const float pi = 3.14159265358979323846f;
395
396 mesh.vertices.reserve((latCount + 1) * (lonCount + 1));
397 mesh.edges.reserve((latCount * lonCount) + ((latCount + 1) * lonCount));
398
399 for (size_t latitude = 0; latitude <= latCount; ++latitude) {
400 const float v =
401 static_cast<float>(latitude) / static_cast<float>(latCount);
402 const float phi = v * pi;
403 const float ringY = radius * std::cos(phi);
404 const float ringRadius = radius * std::sin(phi);
405
406 for (size_t longitude = 0; longitude <= lonCount; ++longitude) {
407 const float u =
408 static_cast<float>(longitude) / static_cast<float>(lonCount);
409 const float theta = u * (2.0f * pi);
410 mesh.vertices.push_back({ringRadius * std::cos(theta), ringY,
411 ringRadius * std::sin(theta)});
412 }
413 }
414
415 const size_t rowSize = lonCount + 1;
416 for (size_t latitude = 0; latitude <= latCount; ++latitude) {
417 for (size_t longitude = 0; longitude < lonCount; ++longitude) {
418 const size_t current = (latitude * rowSize) + longitude;
419 mesh.edges.push_back({current, current + 1});
420 }
421 }
422
423 for (size_t latitude = 0; latitude < latCount; ++latitude) {
424 for (size_t longitude = 0; longitude <= lonCount; ++longitude) {
425 const size_t current = (latitude * rowSize) + longitude;
426 mesh.edges.push_back({current, current + rowSize});
427 }
428 }
429
430 return mesh;
431 }

433 /// Creates a view matrix from camera parameters.
434 static Mat4 lookAt(const Vec3& eye, const Vec3& target, const Vec3& up) {
435 const Vec3 forward = normalize(target - eye);
436 const Vec3 right = normalize(cross(forward, up));
437 const Vec3 actualUp = cross(right, forward);
438
439 Mat4 result = Mat4::identity();
440 result.m[0][0] = right.x;
441 result.m[0][1] = right.y;
442 result.m[0][2] = right.z;
443 result.m[0][3] = -dot(right, eye);
444
445 result.m[1][0] = actualUp.x;
446 result.m[1][1] = actualUp.y;
447 result.m[1][2] = actualUp.z;
448 result.m[1][3] = -dot(actualUp, eye);
449
450 result.m[2][0] = -forward.x;
451 result.m[2][1] = -forward.y;
452 result.m[2][2] = -forward.z;
453 result.m[2][3] = dot(forward, eye);
454 return result;
455 }

457 /// Projects a world-space point into screen space.
458 bool projectPoint(const Vec3& point, int16_t& screenX, int16_t& screenY,
459 float& depth, const Mat4& model = Mat4::identity()) const {
460 ProjectedPoint projected;
461 Vec3 viewPoint = transformAffine(viewMatrix_ * model, point);
462 if (!clipPointToCameraRange(viewPoint)) {
463 return false;
464 }
465 if (!projectViewPoint(viewPoint, projected)) {
466 return false;
467 }
468
469 screenX = projected.x;
470 screenY = projected.y;
471 depth = projected.depth;
472 return true;
473 }

475 /// Renders a wireframe mesh with minimal depth buffering.
476 void renderWireframe(ISurface<RGB_T>& target, const Mesh& mesh,
477 const Mat4& model = Mat4::identity(),
478 RGB_T color = RGB_T(255, 255, 255),
479 bool clearDepth = true) {
480 if (viewportWidth_ != target.width() ||
481 viewportHeight_ != target.height()) {
482 setViewport(target.width(), target.height());
483 }
484 if (clearDepth) {
485 clearDepthBuffer();
486 }
487
488 const Mat4 modelView = viewMatrix_ * model;
489 for (const Edge& edge : mesh.edges) {
490 if (edge.start >= mesh.vertices.size() ||
491 edge.end >= mesh.vertices.size()) {
492 continue;
493 }
494
495 Vec3 start = transformAffine(modelView, mesh.vertices[edge.start]);
496 Vec3 end = transformAffine(modelView, mesh.vertices[edge.end]);
497 if (!clipLineToCameraRange(start, end)) {
498 continue;
499 }
500
501 ProjectedPoint projectedStart;
502 ProjectedPoint projectedEnd;
503 if (!projectViewPoint(start, projectedStart) ||
504 !projectViewPoint(end, projectedEnd)) {
505 continue;
506 }
507
508 drawDepthLine(target, projectedStart, projectedEnd, color);
509 }

607 static Mat4 createPerspectiveMatrix(float fovYDegrees, float aspect,
608 float nearPlane, float farPlane) {
609 Mat4 result;
610 const float halfFovRadians =
611 (fovYDegrees * 3.14159265358979323846f / 180.0f) * 0.5f;
612 const float focalLength = 1.0f / std::tan(halfFovRadians);
613
614 result.m[0][0] = focalLength / aspect;
615 result.m[1][1] = focalLength;
616 result.m[2][2] = (farPlane + nearPlane) / (nearPlane - farPlane);
617 result.m[2][3] = (2.0f * farPlane * nearPlane) / (nearPlane - farPlane);
618 result.m[3][2] = -1.0f;
619 return result;
620 }

622 static Mat4 createOrthographicMatrix(float left, float right, float bottom,
623 float top, float nearPlane,
624 float farPlane) {
625 Mat4 result = Mat4::identity();
626 result.m[0][0] = 2.0f / (right - left);
627 result.m[1][1] = 2.0f / (top - bottom);
628 result.m[2][2] = -2.0f / (farPlane - nearPlane);
629 result.m[0][3] = -(right + left) / (right - left);
630 result.m[1][3] = -(top + bottom) / (top - bottom);
631 result.m[2][3] = -(farPlane + nearPlane) / (farPlane - nearPlane);
632 return result;
633 }

646 static bool clipLineToPlane(Vec3& start, Vec3& end, float planeZ,
647 bool keepLessEqual) {
648 const bool startInside =
649 keepLessEqual ? start.z <= planeZ : start.z >= planeZ;
650 const bool endInside = keepLessEqual ? end.z <= planeZ : end.z >= planeZ;
651
652 if (!startInside && !endInside) {
653 return false;
654 }
655 if (startInside && endInside) {
656 return true;
657 }
658
659 const float deltaZ = end.z - start.z;
660 if (deltaZ == 0.0f) {
661 return false;
662 }
663
664 const float t = (planeZ - start.z) / deltaZ;
665 const Vec3 intersection = start + ((end - start) * t);
666 if (!startInside) {
667 start = intersection;
668 } else {
669 end = intersection;
670 }
671 return true;
672 }

674 bool projectViewPoint(const Vec3& viewPoint,
675 ProjectedPoint& projected) const {
676 const Vec4 clip =
677 projectionMatrix_ * Vec4{viewPoint.x, viewPoint.y, viewPoint.z, 1.0f};
678 if (clip.w == 0.0f) {
679 return false;
680 }
681
682 const float inverseW = 1.0f / clip.w;
683 const float ndcX = clip.x * inverseW;
684 const float ndcY = clip.y * inverseW;
685 const float ndcZ = clip.z * inverseW;
686 const float screenX =
687 ((ndcX * 0.5f) + 0.5f) *
688 static_cast<float>(viewportWidth_ > 0 ? viewportWidth_ - 1 : 0);
689 const float screenY =
690 (1.0f - ((ndcY * 0.5f) + 0.5f)) *
691 static_cast<float>(viewportHeight_ > 0 ? viewportHeight_ - 1 : 0);
692
693 projected.x = static_cast<int16_t>(std::lround(screenX));
694 projected.y = static_cast<int16_t>(std::lround(screenY));
695 projected.depth = (ndcZ + 1.0f) * 0.5f;
696 return true;
697 }
698
699 void drawDepthLine(ISurface<RGB_T>& target, const ProjectedPoint& start,
700 const ProjectedPoint& end, RGB_T color) {
701 const int deltaX = static_cast<int>(end.x) - static_cast<int>(start.x);
702 const int deltaY = static_cast<int>(end.y) - static_cast<int>(start.y);
703 const int steps = std::max(std::abs(deltaX), std::abs(deltaY));
704
705 if (steps == 0) {
706 plotDepthPixel(target, start.x, start.y, start.depth, color);
707 return;
708 }
709
710 for (int step = 0; step <= steps; ++step) {
711 const float factor = static_cast<float>(step) / static_cast<float>(steps);
712 const int x = static_cast<int>(std::lround(
713 static_cast<float>(start.x) + (static_cast<float>(deltaX) * factor)));
714 const int y = static_cast<int>(std::lround(
715 static_cast<float>(start.y) + (static_cast<float>(deltaY) * factor)));
716 const float depth = start.depth + ((end.depth - start.depth) * factor);
717 plotDepthPixel(target, x, y, depth, color);
718 }
719 }
720
721 void plotDepthPixel(ISurface<RGB_T>& target, int x, int y, float depth,
722 RGB_T color) {
723 if (x < 0 || y < 0 || static_cast<size_t>(x) >= viewportWidth_ ||
724 static_cast<size_t>(y) >= viewportHeight_) {
725 return;
726 }
727
728 const size_t bufferIndex =
729 (static_cast<size_t>(y) * viewportWidth_) + static_cast<size_t>(x);
730 if (bufferIndex >= depthBuffer_.size()) {
731 return;
732 }
733 if (depth >= depthBuffer_[bufferIndex]) {
734 return;
735 }
736
737 depthBuffer_[bufferIndex] = depth;
738 target.setPixel(static_cast<size_t>(x), static_cast<size_t>(y), color);
739 }
740};
