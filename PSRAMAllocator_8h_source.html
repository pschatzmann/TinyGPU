20/**
21 * @class PSRAMAllocator
22 * @brief STL-compatible allocator that prefers PSRAM allocation with fallback
23 *
24 * PSRAMAllocator is a custom C++ allocator that attempts to allocate memory in
25 * external PSRAM (using heap_caps_aligned_calloc with MALLOC_CAP_SPIRAM) and
26 * falls back to regular heap allocation if PSRAM allocation fails.
27 *
28 * This allocator is particularly useful for ESP32 platforms with external PSRAM
29 * where large buffers (like video frames) should be stored in PSRAM to preserve
30 * internal SRAM for other uses. For small, frequently-accessed buffers that
31 * need fast access, consider using RAMAllocator instead.
32 *
33 * @tparam T The type of objects to allocate
34 *
35 * @note On ESP-IDF platforms, uses heap_caps_aligned_calloc for PSRAM
36 * allocation
37 * @note On Arduino platforms, the heap_caps functions are shimmed to standard
38 * malloc/free
39 * @note Memory is zero-initialized (calloc behavior)
40 * @note Provides proper alignment for the allocated type
41 *
42 * Example usage:
43 * @code
44 * #include "PSRAMAllocator.h"
45 *
46 * // Use PSRAMAllocator with Vector for large buffers
47 * std::vector<uint8_t, PSRAMAllocator<uint8_t>> largeBuffer;
48 * largeBuffer.resize(1024 * 1024); // Large buffer preferably in PSRAM
49 *
50 * @endcode
51 */

77 /**
78 * @brief Allocate memory for n objects of type T
79 *
80 * Attempts to allocate memory in the following priority order:
81 * 1. PSRAM using heap_caps_aligned_calloc (if available)
82 * 2. Aligned allocation using aligned_alloc (C11)
83 * 3. Standard calloc as fallback
84 *
85 * All allocated memory is zero-initialized.
86 *
87 * @param n Number of objects of type T to allocate
88 * @return Pointer to allocated memory
89 * @throws std::bad_alloc if allocation fails
90 *
91 * @note Memory is aligned to alignof(T)
92 * @note On ESP32 with PSRAM, prefers external PSRAM allocation
93 */
94 T* allocate(std::size_t n) {
95 if (n == 0) return nullptr;
96 // total bytes
97 std::size_t bytes = n * sizeof(T);
98 // Prefer PSRAM: use heap_caps_aligned_calloc if available. Request
99 // alignment equal to alignof(T).
100 void* p = nullptr;
101#if defined(MALLOC_CAP_SPIRAM)
102 // heap_caps_aligned_calloc(alignment, nmemb, size, caps)
103 p = heap_caps_aligned_calloc(alignof(T), n, sizeof(T),
104 (unsigned)MALLOC_CAP_SPIRAM);
105#endif
106 if (!p) {
107 // Fallback: try aligned allocation where available
108#if defined(__STDC_VERSION__) && (__STDC_VERSION__ >= 201112L)
109 // aligned_alloc requires size to be multiple of alignment
110 std::size_t align = alignof(T);
111 std::size_t alloc_size = bytes;
112 if (align > 1) {
113 std::size_t rem = alloc_size % align;
114 if (rem) alloc_size += (align - rem);
115 }
116 p = aligned_alloc(align, alloc_size);
117 if (p) {
118 // zero initialize
119 std::memset(p, 0, bytes);
120 } else
121#endif
122 {
123 // last resort: calloc
124 p = std::calloc(n, sizeof(T));
125 }
126 }
127
128 if (!p) throw std::bad_alloc();
129 return static_cast<T*>(p);
130 }

132 /**
133 * @brief Deallocate memory previously allocated by this allocator
134 *
135 * Frees memory that was allocated by allocate(). Uses heap_caps_free
136 * which works for both PSRAM and regular heap allocations.
137 *
138 * @param p Pointer to memory to deallocate (can be nullptr)
139 * @param n Number of objects (ignored, but required by allocator interface)
140 *
141 * @note Safe to call with nullptr
142 * @note The size parameter n is ignored as heap_caps_free doesn't need it
143 */
144 void deallocate(T* p, std::size_t /*n*/) noexcept {
145 if (!p) return;
146 heap_caps_free(p);
147 }
