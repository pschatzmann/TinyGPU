20/**
21 * @class PSRAMAllocator
22 * @brief STL-compatible allocator that prefers PSRAM allocation with fallback
23 *
24 * PSRAMAllocator is a custom C++ allocator that attempts to allocate memory in
25 * external PSRAM (using heap_caps_aligned_calloc with MALLOC_CAP_SPIRAM) and
26 * falls back to regular heap allocation if PSRAM allocation fails.
27 *
28 * This allocator is particularly useful for ESP32 platforms with external PSRAM
29 * where large buffers (like video frames) should be stored in PSRAM to preserve
30 * internal SRAM for other uses. For small, frequently-accessed buffers that
31 * need fast access, consider using RAMAllocator instead.
32 *
33 * @tparam T The type of objects to allocate
34 *
35 * @note On ESP-IDF platforms, uses heap_caps_aligned_calloc for PSRAM
36 * allocation
37 * @note On Arduino platforms, the heap_caps functions are shimmed to standard
38 * malloc/free
39 * @note Memory is zero-initialized (calloc behavior)
40 * @note Provides proper alignment for the allocated type
41 *
42 * Example usage:
43 * @code
44 * #include "PSRAMAllocator.h"
45 *
46 * // Use PSRAMAllocator with std::vector for large buffers
47 * std::vector<uint8_t, PSRAMAllocator<uint8_t>> largeBuffer;
48 * largeBuffer.resize(1024 * 1024); // Large buffer preferably in PSRAM
49 *
50 * // Or use the convenience alias
51 * PSVec<uint8_t> videoFrame; // Large video frame data in PSRAM
52 * videoFrame.resize(640 * 480 * 3);
53 *
54 * // For small buffers in fast RAM, use RAMAllocator (separate header)
55 * #include "RAMAllocator.h"
56 * RAMVec<uint16_t> lookupTable; // Small lookup table in fast RAM
57 * @endcode
58 */

84 /**
85 * @brief Allocate memory for n objects of type T
86 *
87 * Attempts to allocate memory in the following priority order:
88 * 1. PSRAM using heap_caps_aligned_calloc (if available)
89 * 2. Aligned allocation using aligned_alloc (C11)
90 * 3. Standard calloc as fallback
91 *
92 * All allocated memory is zero-initialized.
93 *
94 * @param n Number of objects of type T to allocate
95 * @return Pointer to allocated memory
96 * @throws std::bad_alloc if allocation fails
97 *
98 * @note Memory is aligned to alignof(T)
99 * @note On ESP32 with PSRAM, prefers external PSRAM allocation
100 */
101 T* allocate(std::size_t n) {
102 if (n == 0) return nullptr;
103 // total bytes
104 std::size_t bytes = n * sizeof(T);
105 // Prefer PSRAM: use heap_caps_aligned_calloc if available. Request
106 // alignment equal to alignof(T).
107 void* p = nullptr;
108#if defined(MALLOC_CAP_SPIRAM)
109 // heap_caps_aligned_calloc(alignment, nmemb, size, caps)
110 p = heap_caps_aligned_calloc(alignof(T), n, sizeof(T),
111 (unsigned)MALLOC_CAP_SPIRAM);
112#endif
113 if (!p) {
114 // Fallback: try aligned allocation where available
115#if defined(__STDC_VERSION__) && (__STDC_VERSION__ >= 201112L)
116 // aligned_alloc requires size to be multiple of alignment
117 std::size_t align = alignof(T);
118 std::size_t alloc_size = bytes;
119 if (align > 1) {
120 std::size_t rem = alloc_size % align;
121 if (rem) alloc_size += (align - rem);
122 }
123 p = aligned_alloc(align, alloc_size);
124 if (p) {
125 // zero initialize
126 std::memset(p, 0, bytes);
127 } else
128#endif
129 {
130 // last resort: calloc
131 p = std::calloc(n, sizeof(T));
132 }
133 }
134
135 if (!p) throw std::bad_alloc();
136 return static_cast<T*>(p);
137 }

139 /**
140 * @brief Deallocate memory previously allocated by this allocator
141 *
142 * Frees memory that was allocated by allocate(). Uses heap_caps_free
143 * which works for both PSRAM and regular heap allocations.
144 *
145 * @param p Pointer to memory to deallocate (can be nullptr)
146 * @param n Number of objects (ignored, but required by allocator interface)
147 *
148 * @note Safe to call with nullptr
149 * @note The size parameter n is ignored as heap_caps_free doesn't need it
150 */
151 void deallocate(T* p, std::size_t /*n*/) noexcept {
152 if (!p) return;
153 heap_caps_free(p);
154 }
