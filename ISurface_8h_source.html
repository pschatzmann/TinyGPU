15/**
16 * @brief Abstract framebuffer and drawing interface for TinyGPU surfaces.
17 *
18 * Implementations expose pixel access, primitive drawing, sprite operations,
19 * and text rendering through a common API.
20 *
21 * @note The coordinate x=0 and y=0 corresponds to the top-left corner of the
22 * surface. The x coordinate increases to the right and the y coordinate
23 * increases downwards.
24 */
25template <typename RGB_T = RGB565>
26class ISurface {
27 public:
28 /// Destroys the framebuffer interface.
29 virtual ~ISurface() = default;
30
31 /// Resizes the framebuffer surface.
32 virtual void resize(size_t newWidth, size_t newHeight) = 0;
33 /// Sets a pixel at the given position.
34 virtual void setPixel(size_t x, size_t y, RGB_T color) = 0;
35 /// Returns the pixel at the given position.
36 virtual RGB_T getPixel(size_t x, size_t y) const = 0;
37 /// Scrolls the framebuffer content by the specified offsets.
38 virtual void scroll(int dx, int dy) = 0;
39 /// Clears the framebuffer with a single color.
40 virtual void clear(RGB_T color = RGB_T(0)) = 0;
41 /// Draws a line between two points.
42 virtual void drawLine(size_t x0, size_t y0, size_t x1, size_t y1,
43 RGB_T color) = 0;
44 /// Draws a rectangle outline.
45 virtual void drawRect(size_t x, size_t y, size_t w, size_t h,
46 RGB_T color) = 0;
47 /// Fills a rectangle.
48 virtual void fillRect(size_t x, size_t y, size_t w, size_t h,
49 RGB_T color) = 0;
50 /// Draws a circle outline.
51 virtual void drawCircle(size_t x, size_t y, size_t r, RGB_T color) = 0;
52 /// Fills a circle.
53 virtual void fillCircle(size_t x, size_t y, size_t r, RGB_T color) = 0;
54 /// Draws a sprite using an optional transparent color.
55 virtual void drawSprite(size_t x, size_t y, const ISurface<RGB_T>& sprite,
56 RGB_T invisibleColor = RGB_T(0)) = 0;
57 /// Clears the area covered by a sprite.
58 virtual void clearSprite(size_t x, size_t y, ISurface<RGB_T>& sprite,
59 RGB_T clearColor = RGB_T(0)) = 0;
60 /// Copies pixels from the framebuffer into a sprite.
61 virtual void copySprite(size_t x, size_t y, const ISurface<RGB_T>& sprite) = 0;
62 /// Returns the framebuffer width in pixels.
63 virtual size_t width() const = 0;
64 /// Returns the framebuffer height in pixels.
65 virtual size_t height() const = 0;
66 /// Draws UTF-8 text using the configured font.
67 virtual void drawText(int16_t x, int16_t y, const char* text,
68 RGB_T foreground, RGB_T background = RGB_T(0),
69 bool opaque = false, uint8_t scale = 1,
70 uint8_t spacing = 1, uint8_t lineSpacing = 1) = 0;
71
72 /// Returns the currently set font for text rendering.
73 virtual IFont<RGB_T>& font() = 0;
74 /// Provides access to the framebuffer as a byte buffer.
75 virtual const uint8_t* data() const = 0;
76 /// Provides the total data size in bytes.
77 virtual size_t size() const = 0;
78};
