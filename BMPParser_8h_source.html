14 * @brief Incremental BMP decoder for TinyGPU framebuffers.
15 *
16 * The parser accepts BMP file bytes through write(), buffers them until enough
17 * data is available, then decodes supported uncompressed BMP formats directly
18 * into the provided ISurface target using RGB565 pixels.
19 */
20template <typename RGB_T = RGB565>

45 size_t write(const uint8_t* data, size_t length) {

47 return 0;

51 tryDecode();
52 return length;

109 void tryDecode() {
110 if (status_ != Status::Collecting) {
111 return;
112 }
113 if (buffer_.size() < 26) {
114 return;
115 }
116
117 HeaderInfo header;
118 if (!parseHeader(header)) {
119 return;
120 }
121
122 const uint32_t requiredSize = requiredDataSize(header);
123 if (requiredSize == 0 || buffer_.size() < requiredSize) {
124 return;
125 }
126
127 if (!decode(header)) {
128 return;
129 }
130
131 status_ = Status::Complete;
132 }

134 bool parseHeader(HeaderInfo& header) {
135 if (buffer_.size() < 54) {
136 return false;
137 }
138
139 if (readU16(0) != 0x4D42) {
140 setError("Invalid BMP signature");
141 return false;
142 }
143
144 header.fileSize = readU32(2);
145 header.pixelOffset = readU32(10);
146 header.dibHeaderSize = readU32(14);
147 if (header.dibHeaderSize < 40) {
148 setError("Unsupported BMP DIB header");
149 return false;
150 }
151
152 if (buffer_.size() < 14 + header.dibHeaderSize) {
153 return false;
154 }
155
156 header.width = readS32(18);
157 header.height = readS32(22);
158 header.planes = readU16(26);
159 header.bitsPerPixel = readU16(28);
160 header.compression = readU32(30);
161 header.imageSize = readU32(34);
162 header.colorsUsed = readU32(46);
163
164 if (header.planes != 1) {
165 setError("Unsupported BMP plane count");
166 return false;
167 }
168 if (header.width <= 0 || header.height == 0) {
169 setError("Invalid BMP dimensions");
170 return false;
171 }
172
173 if (header.bitsPerPixel != 8 && header.bitsPerPixel != 16 &&
174 header.bitsPerPixel != 24 && header.bitsPerPixel != 32) {
175 setError("Unsupported BMP bit depth");
176 return false;
177 }
178
179 if (header.compression != kCompressionRgb &&
180 header.compression != kCompressionBitfields) {
181 setError("Unsupported BMP compression");
182 return false;
183 }
184
185 topDown_ = header.height < 0;
186 imageWidth_ = static_cast<size_t>(header.width);
187 imageHeight_ =
188 static_cast<size_t>(topDown_ ? -header.height : header.height);
189
190 if (!parseBitMasks(header)) {
191 return false;
192 }
193 return parsePalette(header);
194 }
195
196 bool parseBitMasks(HeaderInfo& header) {
197 if (header.bitsPerPixel == 16 || header.bitsPerPixel == 32) {
198 if (header.compression == kCompressionBitfields) {
199 const size_t maskOffset =
200 14 + (header.dibHeaderSize >= 52 ? 40 : header.dibHeaderSize);
201 const size_t requiredMasks = header.dibHeaderSize >= 56 ? 16 : 12;
202 if (buffer_.size() < maskOffset + requiredMasks) {
203 return false;
204 }
205
206 header.redMask = readU32(maskOffset);
207 header.greenMask = readU32(maskOffset + 4);
208 header.blueMask = readU32(maskOffset + 8);
209 if (requiredMasks == 16) {
210 header.alphaMask = readU32(maskOffset + 12);
211 }
212 } else if (header.bitsPerPixel == 16) {
213 header.redMask = 0x7C00;
214 header.greenMask = 0x03E0;
215 header.blueMask = 0x001F;
216 } else {
217 header.redMask = 0x00FF0000;
218 header.greenMask = 0x0000FF00;
219 header.blueMask = 0x000000FF;
220 }
221 }
222
223 return true;
224 }

226 bool parsePalette(const HeaderInfo& header) {
227 palette_.clear();
228 if (header.bitsPerPixel != 8) {
229 return true;
230 }
231
232 const uint32_t paletteEntryCount =
233 header.colorsUsed != 0 ? header.colorsUsed : 256;
234 const size_t paletteOffset = 14 + header.dibHeaderSize;
235 const size_t paletteSize = static_cast<size_t>(paletteEntryCount) * 4;
236 if (buffer_.size() < paletteOffset + paletteSize) {
237 return false;
238 }
239
240 palette_.reserve(paletteEntryCount);
241 for (uint32_t index = 0; index < paletteEntryCount; ++index) {
242 const size_t entryOffset =
243 paletteOffset + (static_cast<size_t>(index) * 4);
244 palette_.push_back(RGB565(buffer_[entryOffset + 2],
245 buffer_[entryOffset + 1],
246 buffer_[entryOffset]));
247 }
248 return true;
249 }

251 uint32_t requiredDataSize(const HeaderInfo& header) const {
252 const uint32_t rowStride = bmpRowStride(header.bitsPerPixel, imageWidth_);
253 const uint32_t computedImageSize =
254 rowStride * static_cast<uint32_t>(imageHeight_);
255 const uint32_t pixelBytes =
256 header.imageSize != 0 ? header.imageSize : computedImageSize;
257 const uint32_t computedSize = header.pixelOffset + pixelBytes;
258
259 if (header.fileSize != 0) {
260 return std::max(header.fileSize, computedSize);
261 }
262 return computedSize;
263 }

265 bool decode(const HeaderInfo& header) {
266 if (header.pixelOffset >= buffer_.size()) {
267 setError("Invalid BMP pixel offset");
268 return false;
269 }
270
271 target_.resize(imageWidth_, imageHeight_);
272 const uint32_t rowStride = bmpRowStride(header.bitsPerPixel, imageWidth_);
273 for (size_t row = 0; row < imageHeight_; ++row) {
274 const size_t sourceRow = topDown_ ? row : (imageHeight_ - 1 - row);
275 const size_t rowOffset = header.pixelOffset + (sourceRow * rowStride);
276 if (rowOffset + rowStride > buffer_.size()) {
277 setError("Incomplete BMP pixel data");
278 return false;
279 }
280
281 if (!decodeRow(header, rowOffset, row)) {
282 return false;
283 }
284 }
285
286 return true;
287 }

289 bool decodeRow(const HeaderInfo& header, size_t rowOffset, size_t targetY) {
290 switch (header.bitsPerPixel) {
291 case 8:
292 return decode8BitRow(rowOffset, targetY);
293 case 16:
294 return decode16BitRow(header, rowOffset, targetY);
295 case 24:
296 return decode24BitRow(rowOffset, targetY);
297 case 32:
298 return decode32BitRow(header, rowOffset, targetY);
299 default:
300 setError("Unsupported BMP row format");
301 return false;
302 }
303 }
304
305 bool decode8BitRow(size_t rowOffset, size_t targetY) {
306 if (palette_.empty()) {
307 setError("Missing BMP palette");
308 return false;
309 }
310
311 for (size_t x = 0; x < imageWidth_; ++x) {
312 const uint8_t paletteIndex = buffer_[rowOffset + x];
313 if (paletteIndex >= palette_.size()) {
314 setError("BMP palette index out of range");
315 return false;
316 }
317 target_.setPixel(x, targetY, palette_[paletteIndex]);
318 }
319 return true;
320 }
321
322 bool decode16BitRow(const HeaderInfo& header, size_t rowOffset,
323 size_t targetY) {
324 for (size_t x = 0; x < imageWidth_; ++x) {
325 const uint16_t pixelValue =
326 static_cast<uint16_t>(buffer_[rowOffset + (x * 2)]) |
327 (static_cast<uint16_t>(buffer_[rowOffset + (x * 2) + 1]) << 8);
328 target_.setPixel(x, targetY, decodeMaskedPixel(pixelValue, header));
329 }
330 return true;
331 }
332
333 bool decode24BitRow(size_t rowOffset, size_t targetY) {
334 for (size_t x = 0; x < imageWidth_; ++x) {
335 const size_t pixelOffset = rowOffset + (x * 3);
336 target_.setPixel(x, targetY,
337 RGB565(buffer_[pixelOffset + 2],
338 buffer_[pixelOffset + 1], buffer_[pixelOffset]));
339 }
340 return true;
341 }
342
343 bool decode32BitRow(const HeaderInfo& header, size_t rowOffset,
344 size_t targetY) {
345 for (size_t x = 0; x < imageWidth_; ++x) {
346 const size_t pixelOffset = rowOffset + (x * 4);
347 const uint32_t pixelValue =
348 static_cast<uint32_t>(buffer_[pixelOffset]) |
349 (static_cast<uint32_t>(buffer_[pixelOffset + 1]) << 8) |
350 (static_cast<uint32_t>(buffer_[pixelOffset + 2]) << 16) |
351 (static_cast<uint32_t>(buffer_[pixelOffset + 3]) << 24);
352 target_.setPixel(x, targetY, decodeMaskedPixel(pixelValue, header));
353 }
354 return true;
355 }

357 RGB565 decodeMaskedPixel(uint32_t pixelValue,
358 const HeaderInfo& header) const {
359 const uint8_t red = extractChannel(pixelValue, header.redMask);
360 const uint8_t green = extractChannel(pixelValue, header.greenMask);
361 const uint8_t blue = extractChannel(pixelValue, header.blueMask);
362 return RGB565(red, green, blue);
363 }
364
365 static uint8_t extractChannel(uint32_t pixelValue, uint32_t mask) {
366 if (mask == 0) {
367 return 0;
368 }
369
370 uint8_t shift = 0;
371 uint8_t bits = 0;
372 while (((mask >> shift) & 0x1U) == 0U) {
373 ++shift;
374 }
375 while (((mask >> (shift + bits)) & 0x1U) != 0U) {
376 ++bits;
377 }
378
379 const uint32_t value = (pixelValue & mask) >> shift;
380 const uint32_t maxValue = (1UL << bits) - 1UL;
381 return maxValue == 0 ? 0 : static_cast<uint8_t>((value * 255UL) / maxValue);
382 }
383
384 static uint32_t bmpRowStride(uint16_t bitsPerPixel, size_t width) {
385 const uint32_t bitsPerRow = static_cast<uint32_t>(width) * bitsPerPixel;
386 return ((bitsPerRow + 31U) / 32U) * 4U;
387 }

389 uint16_t readU16(size_t offset) const {
390 return static_cast<uint16_t>(buffer_[offset]) |
391 (static_cast<uint16_t>(buffer_[offset + 1]) << 8);
392 }

394 uint32_t readU32(size_t offset) const {
395 return static_cast<uint32_t>(buffer_[offset]) |
396 (static_cast<uint32_t>(buffer_[offset + 1]) << 8) |
397 (static_cast<uint32_t>(buffer_[offset + 2]) << 16) |
398 (static_cast<uint32_t>(buffer_[offset + 3]) << 24);
399 }
