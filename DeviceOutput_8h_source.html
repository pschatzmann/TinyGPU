26/**
27 * @brief Common base class for SPI-based display drivers.
28 *
29 * Handles SPI pin setup, hardware reset, address window, and SPI data helpers.
30 * Specific display drivers should inherit from this and implement their own
31 * init sequence.
32 */
33class SPIDisplayDriver : public DisplayDriver {
34 public:
35 SPIDisplayDriver(SPIClass& spi, int8_t cs, int8_t dc, int8_t rst = -1,
36 size_t xOffset = 0, size_t yOffset = 0)
37 : spi_(spi),
38 cs_(cs),
39 dc_(dc),
40 rst_(rst),
41 xOffset_(xOffset),
42 yOffset_(yOffset) {}
43 void setAddressWindow(size_t x, size_t y, size_t w, size_t h) override {
44 writeCommand(0x2A);
45 writeData16(x + xOffset_, x + xOffset_ + w - 1);
46 writeCommand(0x2B);
47 writeData16(y + yOffset_, y + yOffset_ + h - 1);
48 writeCommand(0x2C);
49 }
50 void writeData(ISurface& surface) override {
51 setAddressWindow(0, 0, surface.width(), surface.height());
52 digitalWrite(dc_, HIGH);
53 digitalWrite(cs_, LOW);
54 spi_.writeBytes(surface.data(), surface.size());
55 digitalWrite(cs_, HIGH);
56 }
57
58 protected:
59 SPIClass& spi_;
60 int8_t cs_, dc_, rst_;
61 size_t xOffset_, yOffset_;
62 void setupPinsAndReset() {
63 pinMode(cs_, OUTPUT);
64 pinMode(dc_, OUTPUT);
65 if (rst_ >= 0) {
66 pinMode(rst_, OUTPUT);
67 digitalWrite(rst_, LOW);
68 delay(20);
69 digitalWrite(rst_, HIGH);
70 delay(150);
71 }
72 digitalWrite(cs_, HIGH);
73 }
74 void writeCommand(uint8_t cmd) {
75 digitalWrite(dc_, LOW);
76 digitalWrite(cs_, LOW);
77 spi_.transfer(cmd);
78 digitalWrite(cs_, HIGH);
79 }
80 void writeData16(uint16_t d1, uint16_t d2) {
81 digitalWrite(dc_, HIGH);
82 digitalWrite(cs_, LOW);
83 spi_.transfer(d1 >> 8);
84 spi_.transfer(d1 & 0xFF);
85 spi_.transfer(d2 >> 8);
86 spi_.transfer(d2 & 0xFF);
87 digitalWrite(cs_, HIGH);
88 }
89 void writeData8(uint8_t data) {
90 digitalWrite(dc_, HIGH);
91 digitalWrite(cs_, LOW);
92 spi_.transfer(data);
93 digitalWrite(cs_, HIGH);
94 }
95};

183 * @brief Hardware abstraction for writing framebuffers to SPI-based color

185 *
186 * DeviceOutput provides a unified interface for sending RGB565 framebuffer data
187 * to common embedded display controllers (ST7735, ST7789, ILI9341, HX8357)
188 * using the Arduino SPI API.
189 *
190 * Features:
191 * - Handles X/Y offsets for displays that require them (via driver)
192 * - Supports configurable color order (RGB/BGR) via RGB565
193 * - Initialization sequence and address window logic delegated to driver
194 * - Optional hardware reset pin (via driver)
195 * - Efficient bulk framebuffer transfer
196 *
197 * Usage:
198 * DeviceOutput out(new ILI9341Driver(spi, cs, dc, rst), 240, 320);
199 * out.writeData(framebuffer);
200 *
201 * @note For color order, set RGB565::setColorOrderBGR(true/false) as needed for
202 * your display, before creating any data.

206 * @brief Hardware abstraction for writing framebuffers to SPI-based color

215 * - Supports configurable color order (RGB/BGR) via RGB565
216 * - Initialization sequence and address window logic delegated to driver
217 * - Optional hardware reset pin (via driver)
218 * - Efficient bulk framebuffer transfer
219 *
220 * Usage:
221 * DeviceOutput out(new ILI9341Driver(spi, cs, dc, rst), 240, 320);
222 * out.writeData(framebuffer);
223 *
224 * @note For color order, set RGB565::setColorOrderBGR(true/false) as needed for
225 * your display, before creating any data.
226 */
227class DeviceOutput {
228 public:
229 /**
230 * @brief Construct a DeviceOutput with a display driver.
231 * @param driver Pointer to a display driver (ownership not taken)
232 * @param width Display width in pixels
233 * @param height Display height in pixels
234 */
235 DeviceOutput(DisplayDriver* driver, size_t width, size_t height)
236 : driver_(driver), width_(width), height_(height) {}
237
238 /// Initializes the display and prepares it for receiving data.
239 bool begin() {
240 if (driver_) {
241 driver_->init();
242 return true;
243 }
244 return false;
245 }
246
247 /// Returns the display width in pixels.
248 size_t width() const { return width_; }
249
250 /// Returns the display height in pixels.
251 size_t height() const { return height_; }
252
253 /// Write a framebuffer to the display (w x h, RGB565, 2 bytes per pixel).
254 void writeData(ISurface& surface, size_t x = 0, size_t y = 0) {
255 if (driver_) {
256 // if the surface size doesn't match the display size, we need to copy it
257 // to a temporary buffer that does, because the driver expects a full
258 // framebuffer for the address window.
259 if (surface.width() != width_ || surface.height() != height_) {
260 Sprite temp(surface.width(), surface.height(), surface.font());
261 surface.copySprite(x, y, temp);
262 driver_->writeData(temp);
263 return;
264 }
265 // write as is
266 driver_->writeData(surface);
267 }
268 }
269
270 private:
271 DisplayDriver* driver_ = nullptr;
272 size_t width_;
273 size_t height_;
274 bool is_active = false;

276 // SPI helpers and display-specific logic are now in the driver classes.
277};
